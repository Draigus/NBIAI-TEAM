# ---------- UA Plan
ua = wb.add_worksheet("UA_Plan")
ua.freeze_panes(4, 2)
ua.set_column("A:A", 16)
for c in range(1, 12):
    ua.set_column(c, c, 14)

ua.write(0,0,"UA Plan by channel", fmt_title)
ua.write(2,0,"Month", fmt_head)
channels = ["Telegram Ads","Influencers","Cross-Promo"]
for j,ch in enumerate(channels):
    base_col = 1 + j*3
    ua.write(2, base_col, f"{ch} Budget £", fmt_head)
    ua.write(2, base_col+1, f"{ch} CPI £", fmt_head)
    ua.write(2, base_col+2, f"{ch} Installs", fmt_head)

# Defaults for month 1 and growth formula thereafter
horizon = 60
for r in range(1, horizon+1):
    row = 2 + r
    ua.write_formula(row, 0, f"={r}", fmt_num)
    for j in range(len(channels)):
        c0 = 1 + j*3
        if r==1:
            ua.write(row, c0, 50000, fmt_blue_cur)
            ua.write(row, c0+1, 0.55, fmt_blue_cur)
        else:
            ua.write_formula(row, c0, f"={xlsxwriter.utility.xl_rowcol_to_cell(row-1, c0)}*(1+UA_BUDGET_GROWTH)", fmt_cur2)
            ua.write_formula(row, c0+1, f"={xlsxwriter.utility.xl_rowcol_to_cell(row-1, c0+1)}*(1+CPI_GROWTH)", fmt_cur2)
        ua.write_formula(row, c0+2, f"={xlsxwriter.utility.xl_rowcol_to_cell(row, c0)}/{xlsxwriter.utility.xl_rowcol_to_cell(row, c0+1)}", fmt_num)

# Totals and referrals
last_row = 2 + horizon
ua.write(2, 10, "Paid installs", fmt_head)
ua.write(2, 11, "Referral installs", fmt_head)
ua.write(2, 12, "Organic baseline", fmt_head)
ua.write(2, 13, "Total installs", fmt_head)

for r in range(1, horizon+1):
    row = 2 + r
    paid_installs_sum = "+".join([f"{xlsxwriter.utility.xl_rowcol_to_cell(row, 1 + j*3 + 2)}" for j in range(len(channels))])
    ua.write_formula(row, 10, f"={paid_installs_sum}", fmt_num)
    # Referral installs = K_FACTOR * previous month's paid installs
    if r==1:
        ua.write(row, 11, 0, fmt_num)
    else:
        ua.write_formula(row, 11, f"=K{row-1}*K_FACTOR", fmt_num)
    # Organic baseline sized so paid share equals PAID_SHARE on a paid+organic basis
    # paid = PAID_SHARE * total_base ⇒ organic_base = paid*(1-PAID_SHARE)/PAID_SHARE
    ua.write_formula(row, 12, f"=IF(PAID_SHARE>0, K{row}*(1-PAID_SHARE)/PAID_SHARE, 0)", fmt_num)
    ua.write_formula(row, 13, f"=K{row}+L{row}+M{row}", fmt_num)

# Define named range for installs vector
wb.define_name("UA_NEW_INSTALLS", f"='UA_Plan'!$N$4:$N${last_row}")

# ---------- Forecast (driven by UA installs)
fc = wb.add_worksheet("Forecast")
fc.freeze_panes(3, 2)
fc.set_column("A:A", 10)
for col in range(1, 22):
    fc.set_column(col, col, 14)
fc.write(0, 0, "Monthly Operating Forecast", fmt_title)

headers = [
    "Month", "New installs", "Paid installs", "Organic installs",
    "MAU start", "Churn", "Remaining MAU", "Adds", "MAU end",
    "DAU", "Payers Other IAP", "Other IAP £", "Battle Pass £",
    "Commander £", "Cosmetic £", "Tourney take £",
    "Gross Revenue £"
]
for c,h in enumerate(headers):
    fc.write(2, c, h, fmt_head)

for r in range(1, horizon+1):
    row = 2 + r
    fc.write_formula(row, 0, f"={r}", fmt_num)
    # Installs from UA_Plan
    fc.write_formula(row, 1, f"=INDEX(UA_NEW_INSTALLS, A{row})", fmt_num)
    # Paid installs from UA_Plan col K
    fc.write_formula(row, 2, f"=UA_Plan!K{row}", fmt_num)
    # Organic installs = total - paid
    fc.write_formula(row, 3, f"=B{row}-C{row}", fmt_num)
    # MAU start
    if r==1:
        fc.write_formula(row, 4, "=MONTH1_MAU", fmt_num)
    else:
        fc.write_formula(row, 4, f"=I{row-1}", fmt_num)
    # Churn and remaining
    fc.write_formula(row, 5, f"=E{row}*CHURN", fmt_num)
    fc.write_formula(row, 6, f"=E{row}-F{row}", fmt_num)
    # Adds approximated as new installs
    fc.write_formula(row, 7, f"=B{row}", fmt_num)
    # MAU end
    fc.write_formula(row, 8, f"=G{row}+H{row}", fmt_num)
    # DAU
    fc.write_formula(row, 9, f"=I{row}*ACTIVITY", fmt_num)
    # Payers (Other IAP) and revenue
    fc.write_formula(row, 10, f"=I{row}*CONV_IAP_OTHER", fmt_num)
    fc.write_formula(row, 11, f"=K{row}*ARPPU_OTHER", fmt_cur2)
    # Passes and cosmetics
    fc.write_formula(row, 12, f"=I{row}*BP_ATTACH*BP_PRICE", fmt_cur2)
    fc.write_formula(row, 13, f"=IF(A{row}>=3, I{row}*CP_ATTACH*CP_PRICE, 0)", fmt_cur2)
    fc.write_formula(row, 14, f"=I{row}*COS_ATTACH*COS_PRICE", fmt_cur2)
    # Tournament take
    fc.write_formula(row, 15, f"=I{row}*T_GMV_PER_MAU*T_TAKE", fmt_cur2)
    # Gross revenue (sum of exclusive streams)
    fc.write_formula(row, 16, f"=L{row}+M{row}+N{row}+O{row}+P{row}", fmt_cur2)

# Create a downloadable VBA module that finds and breaks external links in an Excel workbook.
# It reports all locations (cells, names, validation, conditional formatting, charts, pivots, shapes, hyperlinks, connections)
# and offers two actions: Break Excel links and Convert external formulas to values.

# Output: /mnt/data/LinkSweeper.bas

from textwrap import dedent
code = dedent(r'''
Attribute VB_Name = "LinkSweeper"
Option Explicit

' Entry points:
'   LinkSweeper_Report    -> creates a "Link_Report" sheet listing every external reference
'   LinkSweeper_BreakAll  -> breaks Excel external links and deletes external Named Ranges
'   LinkSweeper_Values    -> converts any cells with external workbook formulas to values

Private Const HDR1 As String = "Object"
Private Const HDR2 As String = "Location"
Private Const HDR3 As String = "Detail"

Public Sub LinkSweeper_Report()
    Dim wb As Workbook: Set wb = ActiveWorkbook
    Dim wsOut As Worksheet
    On Error Resume Next
    Set wsOut = wb.Worksheets("Link_Report")
    On Error GoTo 0
    If wsOut Is Nothing Then
        Set wsOut = wb.Worksheets.Add(After:=wb.Worksheets(wb.Worksheets.Count))
        wsOut.Name = "Link_Report"
    Else
        wsOut.Cells.Clear
    End If
    
    Dim r As Long: r = 1
    wsOut.Cells(r, 1).Value = "External link report"
    wsOut.Cells(r, 1).Font.Bold = True
    r = r + 2
    wsOut.Cells(r, 1).Value = HDR1
    wsOut.Cells(r, 2).Value = HDR2
    wsOut.Cells(r, 3).Value = HDR3
    wsOut.Rows(r).Font.Bold = True
    r = r + 1
    
    Application.ScreenUpdating = False
    
    ' 1) Workbook LinkSources
    On Error Resume Next
    Dim links As Variant
    links = wb.LinkSources(xlExcelLinks)
    On Error GoTo 0
    If Not IsEmpty(links) Then
        Dim i As Long
        For i = LBound(links) To UBound(links)
            wsOut.Cells(r, 1).Value = "Workbook Link"
            wsOut.Cells(r, 2).Value = "(global)"
            wsOut.Cells(r, 3).Value = CStr(links(i))
            r = r + 1
        Next i
    End If
    
    ' 2) Named ranges
    Dim nm As Name
    For Each nm In wb.Names
        On Error Resume Next
        If InStr(1, nm.RefersTo, "[", vbTextCompare) > 0 Then
            wsOut.Cells(r, 1).Value = "Named Range"
            wsOut.Cells(r, 2).Value = nm.Name
            wsOut.Cells(r, 3).Value = nm.RefersTo
            r = r + 1
        End If
        On Error GoTo 0
    Next nm
    
    ' 3) Worksheets
    Dim ws As Worksheet
    For Each ws In wb.Worksheets

                On Error Resume Next
                Dim f As String: f = CStr(c.Formula)
                On Error GoTo 0
                If InStr(1, f, "[", vbTextCompare) > 0 Then
                    wsOut.Cells(r, 1).Value = "Cell formula"
                    wsOut.Cells(r, 2).Value = ws.Name & "!" & c.Address(False, False)
                    wsOut.Cells(r, 3).Value = Left$(f, 1000)
                    r = r + 1
                End If
            Next c
        End If
        
        ' b) Data validation
        On Error Resume Next

                    wsOut.Cells(r, 1).Value = "Data validation"
                    wsOut.Cells(r, 2).Value = ws.Name & "!" & v.Address(False, False)
                    wsOut.Cells(r, 3).Value = "F1=" & f1 & "  F2=" & f2
                    r = r + 1
                End If
            Next v
        End If
        
        ' c) Conditional formatting
        Dim fc As FormatCondition
        For Each fc In ws.Cells.FormatConditions
            On Error Resume Next
            Dim frm As String
            frm = fc.Formula1
            On Error GoTo 0
            If InStr(1, frm, "[", vbTextCompare) > 0 Then
                wsOut.Cells(r, 1).Value = "Cond. formatting"
                wsOut.Cells(r, 2).Value = ws.Name
                wsOut.Cells(r, 3).Value = frm
                r = r + 1
            End If
        Next fc
        
        ' d) Hyperlinks
        Dim h As Hyperlink
        For Each h In ws.Hyperlinks
            If InStr(1, h.Address, "[", vbTextCompare) > 0 Then
                wsOut.Cells(r, 1).Value = "Hyperlink"
                wsOut.Cells(r, 2).Value = ws.Name & "!" & h.Range.Address(False, False)
                wsOut.Cells(r, 3).Value = h.Address
                r = r + 1
            End If
        Next h
        
        ' e) Chart series
        Dim ch As ChartObject
        For Each ch In ws.ChartObjects

            For Each s In ch.Chart.SeriesCollection
                On Error Resume Next
                Dim sf As String: sf = s.Formula
                On Error GoTo 0
                If InStr(1, sf, "[", vbTextCompare) > 0 Then
                    wsOut.Cells(r, 1).Value = "Chart series"
                    wsOut.Cells(r, 2).Value = ws.Name & "!" & ch.Name
                    wsOut.Cells(r, 3).Value = sf
                    r = r + 1
                End If
            Next s
        Next ch
        
        ' f) Linked pictures and OLE objects
        Dim shp As Shape
        For Each shp In ws.Shapes
            On Error Resume Next
            If shp.Type = msoLinkedPicture Or shp.Type = msoLinkedOLEObject Then
                wsOut.Cells(r, 1).Value = "Linked shape"
                wsOut.Cells(r, 2).Value = ws.Name & "!" & shp.Name
                wsOut.Cells(r, 3).Value = shp.LinkFormat.SourceFullName
                r = r + 1
            End If
            On Error GoTo 0
        Next shp
    Next ws
    
    ' 4) Pivot caches
    Dim pc As PivotCache
    For Each pc In wb.PivotCaches
        On Error Resume Next
        Dim src As Variant
        src = pc.SourceData
        On Error GoTo 0
        If VarType(src) = vbString Then
            If InStr(1, CStr(src), "[", vbTextCompare) > 0 Then
                wsOut.Cells(r, 1).Value = "Pivot cache"
                wsOut.Cells(r, 2).Value = "(global)"
                wsOut.Cells(r, 3).Value = CStr(src)
                r = r + 1
            End If
        End If
    Next pc
    
    ' 5) Workbook connections
    Dim cn As WorkbookConnection
    For Each cn In wb.Connections
        Dim desc As String: desc = ""
        On Error Resume Next
        desc = cn.Description
        On Error GoTo 0
        wsOut.Cells(r, 1).Value = "Connection"
        wsOut.Cells(r, 2).Value = cn.Name
        wsOut.Cells(r, 3).Value = desc
        r = r + 1
    Next cn
    
    wsOut.Columns("A:C").EntireColumn.AutoFit
    Application.ScreenUpdating = True
    MsgBox "Link report complete. See the 'Link_Report' sheet.", vbInformation
End Sub

Public Sub LinkSweeper_BreakAll()
    Dim wb As Workbook: Set wb = ActiveWorkbook
    Dim links As Variant
    On Error Resume Next
    links = wb.LinkSources(xlExcelLinks)
    On Error GoTo 0
    If Not IsEmpty(links) Then
        Dim i As Long
        For i = LBound(links) To UBound(links)
            On Error Resume Next
            wb.BreakLink Name:=links(i), Type:=xlLinkTypeExcelLinks
            On Error GoTo 0
        Next i
    End If
    
    ' Delete Named Ranges that still point outside
    Dim nm As Name, delList As Collection: Set delList = New Collection
    For Each nm In wb.Names
        On Error Resume Next
        If InStr(1, nm.RefersTo, "[", vbTextCompare) > 0 Then
            delList.Add nm.Name
        End If
        On Error GoTo 0
    Next nm
    Dim k As Long
    For k = 1 To delList.Count
        On Error Resume Next
        wb.Names(delList(k)).Delete
        On Error GoTo 0
    Next k
    
    MsgBox "Attempted to break Excel links and delete external named ranges. Run LinkSweeper_Report again to verify.", vbInformation
End Sub

Public Sub LinkSweeper_Values()
    ' Converts any cell formulas that reference external workbooks to static values
    Dim wb As Workbook: Set wb = ActiveWorkbook
    Dim ws As Worksheet
    For Each ws In wb.Worksheets
        On Error Resume Next
        Dim rngF As Range: Set rngF = ws.UsedRange.SpecialCells(xlCellTypeFormulas)
        On Error GoTo 0
        If Not rngF Is Nothing Then
            Dim c As Range
            For Each c In rngF
                If InStr(1, c.Formula, "[", vbTextCompare) > 0 Then
                    c.Value = c.Value
                End If
            Next c
        End If
    Next ws
    MsgBox "External workbook formulas were converted to values. Run LinkSweeper_Report to verify nothing remains.", vbInformation
End Sub
