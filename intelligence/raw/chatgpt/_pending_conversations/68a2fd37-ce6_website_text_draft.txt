import matplotlib.pyplot as plt
from matplotlib.patches import Rectangle

def draw_wireframe(layout_name, width, height, sections):
    fig, ax = plt.subplots(figsize=(width/100, height/100), dpi=100)
    ax.set_xlim(0, width)
    ax.set_ylim(0, height)
    ax.axis('off')
    
    y = height
    for s in sections:
        h = s['h']
        y -= h
        rect = Rectangle((s.get('x', 0), y), s.get('w', width), h, fill=False, linewidth=1.5)
        ax.add_patch(rect)
        ax.text(s.get('x', 0)+10, y + h - 20, s['label'], fontsize=10, va='top')
        # Optional inner columns
        if 'cols' in s:
            col_x = s.get('x', 0)
            for col in s['cols']:
                ch = h - 20
                cx = col_x
                cw = col['w']
                inner = Rectangle((cx, y+10), cw, ch, fill=False, linestyle='dashed', linewidth=1)
                ax.add_patch(inner)
                ax.text(cx+8, y+ch-8, col['label'], fontsize=8, va='top')
                col_x += cw + col.get('gap', 0)
    fig.tight_layout()
    path = f"/mnt/data/{layout_name}_wireframe.png"
    plt.savefig(path, bbox_inches='tight')
    plt.close(fig)
    return path
