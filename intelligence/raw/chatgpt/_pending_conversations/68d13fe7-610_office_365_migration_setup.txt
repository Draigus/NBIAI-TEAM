```powershell

# For each group, list Planner plans and tasks

  $plans = Get-MgGroupPlannerPlan -GroupId $g.Id -ErrorAction SilentlyContinue
  foreach ($p in $plans) {
    $tasks = Get-MgPlannerPlanTask -PlannerPlanId $p.Id
    [pscustomobject]@{
      GroupId = $g.Id; GroupName = $g.DisplayName
      PlanId  = $p.Id; PlanTitle = $p.Title
      TaskCount = $tasks.Count
    }
  }
}
$result | Export-Csv ".\nsiteam_plans.csv" -NoTypeInformation
```
