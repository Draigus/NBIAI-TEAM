- UI: Minimal Next.js front end for sessions, entities, and chat.

- Exports: Markdown, JSON, PDF, Foundry journals; OneDrive via rclone.

## 4) Core models
`backend/app/models.py`
```python
from sqlalchemy.orm import declarative_base, relationship, Mapped, mapped_column
from sqlalchemy import String, Text, Integer, ForeignKey, DateTime, func, JSON
from sqlalchemy.dialects.postgresql import VECTOR

Base = declarative_base()

class Session(Base):
    __tablename__ = "sessions"
    id: Mapped[int] = mapped_column(primary_key=True)
    title: Mapped[str] = mapped_column(String(200))
    date: Mapped[DateTime] = mapped_column(DateTime, server_default=func.now())
    recap_md: Mapped[str | None] = mapped_column(Text)
    notes: Mapped[list["Note"]] = relationship("Note", back_populates="session")

class Entity(Base):
    __tablename__ = "entities"
    id: Mapped[int] = mapped_column(primary_key=True)
    kind: Mapped[str] = mapped_column(String(32))  # character, location, item, faction, quest
    name: Mapped[str] = mapped_column(String(160), index=True)
    summary: Mapped[str | None] = mapped_column(Text)
    data: Mapped[dict | None] = mapped_column(JSON)

class Note(Base):
    __tablename__ = "notes"
    id: Mapped[int] = mapped_column(primary_key=True)
    session_id: Mapped[int | None] = mapped_column(ForeignKey("sessions.id"))
    text: Mapped[str] = mapped_column(Text)
    chunk: Mapped[str | None] = mapped_column(Text)
    embedding: Mapped[list[float] | None] = mapped_column(VECTOR(768))
    fts: Mapped[str | None] = mapped_column(Text)  # tsvector will be built via trigger
    session: Mapped["Session"] = relationship("Session", back_populates="notes")
```

## 5) Chunking and embeddings
`backend/app/rag/chunking.py`
```python
def split_markdown(md: str, max_chars: int = 1200, overlap: int = 150):
    chunks = []
    start = 0
    while start < len(md):
        end = min(len(md), start + max_chars)
        chunk = md[start:end]
        chunks.append(chunk)
        start = end - overlap
        if start < 0:
            start = 0
        if end == len(md):
            break
    return chunks
```

```python
import httpx, os
from ..settings import settings

async def embed_texts(texts: list[str]) -> list[list[float]]:
    if settings.USE_OPENAI and settings.OPENAI_API_KEY:
        # simple example using text-embedding-3-large
        import openai
        openai.api_key = settings.OPENAI_API_KEY
        resp = await openai.Embedding.acreate(model="text-embedding-3-large", input=texts)

            r.raise_for_status()
            return r.json()["embeddings"]
```

## 6) Hybrid retrieval
`backend/app/rag/retrieve.py`
```python
from sqlalchemy import text
from ..db import sessionmaker
from ..settings import settings

def hybrid_search(query: str, k: int = 12):
    # 1) dense vector from embeddings service
    from .embed import embed_texts
    import anyio
    emb = anyio.run(embed_texts, [query])[0]

    # 2) vector search + FTS combined rank
    sql = text("""
    with q as (select :emb::vector as e)
    select id, session_id, text,
           (embedding <=> (select e from q)) as dist,
           ts_rank_cd(to_tsvector('english', text), plainto_tsquery(:q)) as fts_rank
    from notes
    order by (0.5 * (1.0 - dist) + 0.5 * fts_rank) desc
    limit :k;
    """)
    with sessionmaker() as s:
        rows = s.execute(sql, {"emb": emb, "q": query, "k": k}).mappings().all()
    return rows
```

## 7) Recap generation
`backend/app/services/recap_service.py`
```python
from ..rag.retrieve import hybrid_search
from ..settings import settings
import httpx

PROMPT = """You are a campaign scribe. Write a concise session recap with:
- Title
- Summary in 6–10 sentences
- Key NPCs, Locations, Items
- Decisions and Consequences
Use only the provided notes.

Notes:
{notes}
"""

async def generate_recap(session_id: int):
    rows = hybrid_search(f"session {session_id} recap", k=20)
    notes_text = "\n\n".join(r["text"] for r in rows)
    prompt = PROMPT.format(notes=notes_text)

    if settings.USE_OPENAI and settings.OPENAI_API_KEY:
        import openai, asyncio
        openai.api_key = settings.OPENAI_API_KEY
        resp = await openai.ChatCompletion.acreate(
            model="gpt-4o-mini",
            messages=[{"role":"user","content":prompt}],
            temperature=0.3,
        )
        return resp["choices"][0]["message"]["content"]
    else:
        # call llama.cpp server
        async with httpx.AsyncClient(timeout=180) as client:
            r = await client.post(settings.LOCAL_LLM_URL, json={"prompt": prompt, "n_predict": 512})
            r.raise_for_status()
            return r.text
```

## 8) Transcription route
`backend/app/routes/asr.py`
```python
from fastapi import APIRouter, UploadFile
from faster_whisper import WhisperModel
from ..settings import settings
import tempfile, os

router = APIRouter(prefix="/asr", tags=["asr"])
model = WhisperModel(settings.ASR_MODEL_SIZE, device="cpu", compute_type="int8")

@router.post("/transcribe")
async def transcribe(file: UploadFile):
    with tempfile.NamedTemporaryFile(delete=False, suffix=file.filename) as tmp:
        tmp.write(await file.read())
        tmp.flush()
        segments, _ = model.transcribe(tmp.name, vad_filter=True)
    text = " ".join(seg.text.strip() for seg in segments)
    os.unlink(tmp.name)
    return {"text": text}
```

## 9) Export service
`backend/app/services/export_service.py`

```python
import os, json
from ..settings import settings
from fpdf import FPDF

def export_markdown(md: str, base: str):
    path = os.path.join(settings.EXPORT_DIR, f"{base}.md")
    os.makedirs(settings.EXPORT_DIR, exist_ok=True)
    with open(path, "w", encoding="utf-8") as f:
        f.write(md)
    return path

def export_json(obj: dict, base: str):
    path = os.path.join(settings.EXPORT_DIR, f"{base}.json")
    with open(path, "w", encoding="utf-8") as f:
        json.dump(obj, f, ensure_ascii=False, indent=2)
    return path

def export_pdf(md: str, base: str):
    pdf = FPDF()
    pdf.add_page()
    pdf.set_font("Arial", size=11)
    for line in md.splitlines():
        pdf.multi_cell(0, 6, line)
    path = os.path.join(settings.EXPORT_DIR, f"{base}.pdf")
    pdf.output(path)
    return path
```

## 10) Discord bot ingestion
`discord-bot/bot.py`
```python
import discord, aiohttp, os

TOKEN = os.getenv("DISCORD_TOKEN")
API = os.getenv("API_BASE", "http://localhost:8000")

intents = discord.Intents.default()
intents.message_content = True
client = discord.Client(intents=intents)

@client.event
async def on_message(message):
    if message.author.bot:
        return
    if message.content.startswith("/archive"):
        async with aiohttp.ClientSession() as s:
            payload = {"channel": message.channel.id, "message_id": message.id, "text": message.content}
            await s.post(f"{API}/ingest/discord", json=payload)
        await message.channel.send("Archived.")

client.run(TOKEN)
```
