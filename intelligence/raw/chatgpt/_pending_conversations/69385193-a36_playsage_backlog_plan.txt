def read_docx_text(path, max_chars=20000):
    with zipfile.ZipFile(path) as z:
        xml = z.read('word/document.xml').decode('utf-8')
    # simple regex to drop tags
    text = re.sub(r'</w:p>', '\n', xml)
    text = re.sub(r'<.*?>', '', text)
    text = re.sub(r'\n+', '\n', text)
    if max_chars:
        text = text[:max_chars]
    return text
